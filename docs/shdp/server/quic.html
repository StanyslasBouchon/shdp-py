class ShdpQuicServer(IShdpServer[QuicConnectionProtocol]):
    """QUIC implementation of the SHDP server protocol.
    
    This class implements a QUIC server that handles SHDP protocol connections.
    It manages client connections and provides methods for server operations.
    
    Attributes:
        _server (Optional[QuicConnectionProtocol]): The QUIC server instance
        _clients (Dict[str, QuicConnectionProtocol]): Dictionary of connected clients
        
    Example:
        >>> result = await ShdpQuicServer.listen(15150)
        >>> if result.is_ok():
        ...     server = result.unwrap()
        ...     # Server is now listening
    """

    def __init__(self, cert_path: Path, key_path: Path):
        """Initialize a new SHDP QUIC server instance."""
        self._server: Optional[QuicConnectionProtocol] = None
        self._clients: Dict[str, QuicConnectionProtocol] = {}
        self._cert_path = cert_path
        self._key_path = key_path


    @staticmethod
    async def listen(cert_path: Path, key_path: Path, port: int = 15150) -> Result['ShdpQuicServer', Error]:
        """Start listening for QUIC connections on the specified port.
        
        Args:
            port: Port number to listen on, defaults to 15150
            
        Returns:
            Result['ShdpQuicServer', Error]: Ok with server instance if successful,
                                           Err with error details if failed
                                           
        Example:
            >>> result = await ShdpQuicServer.listen(8080)
            >>> if result.is_ok():
            ...     server = result.unwrap()
        """
        try:
            with open(cert_path, 'rb') as f:
                cert_data = f.read()
            with open(key_path, 'rb') as f:
                key_data = f.read()

            config = QuicConfiguration(
                alpn_protocols=["shdp"],
                is_client=False,
                verify_mode=False,
                certificate=cert_data,
                private_key=key_data,
                cipher_suites=[0x1301, 0x1302, 0x1303],
                supported_versions=[0x0304],
                max_datagram_frame_size=1200
            )
            
            shdp_server = ShdpQuicServer(cert_path, key_path)
            
            quic_server = await serve(
                host='0.0.0.0',
                port=port,
                configuration=config,
                create_protocol=lambda *args, **kwargs: shdp_server._create_protocol(*args, **kwargs)
            )
            
            shdp_server._server = quic_server
            logging.info(f"SHDP QUIC server listening on port {port}")
            return Result.Ok(shdp_server)
            
        except Exception as e:
            return Result.Err(Error.new(ErrorKind.USER_DEFINED, str(e)))
    
    def _create_protocol(self, *args, **kwargs) -> QuicConnectionProtocol:
        """Create a new QUIC protocol instance for a connection.
        
        Returns:
            QuicConnectionProtocol: New protocol instance
        """
        protocol = QuicConnectionProtocol(*args, **kwargs)
        asyncio.create_task(self._accept(protocol))
        return protocol

    async def _accept(self, connection: QuicConnectionProtocol) -> Result[None, Error]:
        """Accept and store a new client connection.
        
        Args:
            connection: The QUIC connection protocol instance
            
        Returns:
            Result[None, Error]: Ok(None) if accepted, Err if failed
        """
        try:
            # Attendre que le transport soit initialisé
            while connection._transport is None:
                await asyncio.sleep(0.1)
            
            client_addr = f"{connection._transport._address[0]}:{connection._transport._address[1]}"
            self._clients[client_addr] = connection
            logging.info(f"Accepted connection from {client_addr}")
            return Result.Ok(None)
        except Exception as e:
            return Result.Err(Error.new(ErrorKind.EXPECTATION_FAILED, str(e)))
    
    async def close(self) -> Result[None, Error]:
        """Close the server and all client connections.
        
        Returns:
            Result[None, Error]: Ok(None) if closed successfully, Err if failed
            
        Example:
            >>> await server.close()
        """
        try:
            if self._server is None:
                return Result.Err(Error.new(ErrorKind.SERVICE_UNAVAILABLE, "Server not listening"))
            
            # Close all client connections
            for client in self._clients.values():
                client.close()
            self._clients.clear()
            
            # Close the server
            self._server.close()
            self._server = None
            
            logging.info("SHDP QUIC server closed")
            return Result.Ok(None)
            
        except Exception as e:
            return Result.Err(Error.new(ErrorKind.USER_DEFINED, str(e)))
    
    def get_clients(self) -> Result[Dict[str, QuicConnectionProtocol], Error]:
        """Get a dictionary of all connected clients.
        
        Returns:
            Result[Dict[str, QuicConnectionProtocol], Error]: Ok with clients dict if successful,
                                                            Err if failed
                                                            
        Example:
            >>> result = server.get_clients()
            >>> if result.is_ok():
            ...     clients = result.unwrap()
            ...     for addr, client in clients.items():
            ...         print(f"Client connected from {addr}")
        """
        return Result.Ok(self._clients)
